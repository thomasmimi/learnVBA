Sub 插入标题7内容()
Dim a As String
Dim i As Integer
Dim j As Integer
For i = 1 To 144
    Selection.Find.text = "代码转换模块"
    Selection.Find.Execute
    Selection.MoveRight Unit:=wdCharacter, Count:=1
        With Selection
			.MoveDown Unit:=wdLine, Count:=2, Extend:=wdExtend
		End With
        With Selection.Find
                .ClearFormatting
				.Text = "主体身份代码，"
                .Replacement.ClearFormatting
                .Replacement.Text = ""
                .Execute Replace:=wdReplaceAll, Forward:=True,Wrap:=wdFindContinue
        End With
		Selection.MoveRight Unit:=wdCharacter, Count:=1
Next
End Sub
